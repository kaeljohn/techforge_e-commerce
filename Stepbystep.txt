1:
git clone <your-github-repo-url>
cd techforge_e-commerce

# Install PHP dependencies
composer install

# Install Javascript/CSS dependencies
npm install

2:
cp .env.example .env

3:
php artisan key:generate

4: (.env)
DB_CONNECTION=mysql
DB_HOST=127.0.0.1
DB_PORT=3306
DB_DATABASE=techforge_e-commerce
DB_USERNAME=root
DB_PASSWORD=

5:
php artisan migrate:fresh --seed
php artisan storage:link

6:
php artisan serve

7:
npm run dev
